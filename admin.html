Firebase Database Rules

Paste rules berikut di Firebase Console → Realtime Database → Rules. 
{
  "rules": {
    "system_settings": {".read": true,".write": "auth != null && root.child('users_config').child(auth.uid).child('role').val() === 'OWNER'"},
    "role_limit": {".read": true,".write": "auth != null && root.child('users_config').child(auth.uid).child('role').val() === 'OWNER'"},
    "user_ratings": {".read": true, ".write": true},
    "users_config": {"$uid": {".read": "auth != null && (auth.uid === $uid || root.child('users_config').child(auth.uid).child('role').val() === 'OWNER' || root.child('users_config').child(auth.uid).child('role').val() === 'ADMIN')",".write": "auth != null && (auth.uid === $uid || root.child('users_config').child(auth.uid).child('role').val() === 'OWNER' || root.child('users_config').child(auth.uid).child('role').val() === 'ADMIN')"}},
    "daily_usage": {"$uid": {".read": "auth != null && auth.uid === $uid || $uid.matches('^GUEST_.*$')",".write": "auth != null && auth.uid === $uid || $uid.matches('^GUEST_.*$')"}},
    "user_sessions": {"$uid": {".read": "auth != null && auth.uid === $uid || $uid.matches('^GUEST_.*$')",".write": "auth != null && auth.uid === $uid || $uid.matches('^GUEST_.*$')"}},
    "archived_sessions": {".read": "auth != null && (root.child('users_config').child(auth.uid).child('role').val() === 'OWNER' || root.child('users_config').child(auth.uid).child('role').val() === 'ADMIN')",".write": "auth != null && (root.child('users_config').child(auth.uid).child('role').val() === 'OWNER' || root.child('users_config').child(auth.uid).child('role').val() === 'ADMIN')"},
    "analytics": {".read": "auth != null && (root.child('users_config').child(auth.uid).child('role').val() === 'OWNER' || root.child('users_config').child(auth.uid).child('role').val() === 'ADMIN')","guests": {"$guestId": {".write": true}},"traffic": {"$uid": {".write": "auth != null && auth.uid === $uid"}}}
  }
}
